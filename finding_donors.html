def train_predict(learner, sample_size, X_train, y_train, X_test, y_test): 
    '''
    inputs:
       - learner: the learning algorithm to be trained and predicted on
       - sample_size: the size of samples (number) to be drawn from training set
       - X_train: features training set
       - y_train: income training set
       - X_test: features testing set
       - y_test: income testing set
    '''
    
    results = {}
    
    # TODOsample_size
    X_train = X_train[0:int(sample_size)]
    y_train = y_train[0:int(sample_size)]
    # TODO: Fit the learner to the training data using slicing with 'sample_size'
    start = time() # 
    learner = learner.fit(X_train,y_train)
    end = time() # 
    
    # TODO
    results['train_time'] = end-start
    
    # TODO: 
    #       300
    start = time() # 
    predictions_test = learner.predict(X_test)
    predictions_train = learner.predict(X_train[0:300])
    end = time() # 
    
    # TODO
    results['pred_time'] = end-start
            
    # TODO300
    results['acc_train'] = accuracy_score(y_train[0:300],predictions_train)
        
    # TODO
    results['acc_test'] = accuracy_score(y_test,predictions_test)
    
    # TODO300F-score
    results['f_train'] = fbeta_score(y_train[0:300],predictions_train, beta=0.5)
        
    # TODOF-score
    results['f_test'] = fbeta_score(y_test,predictions_test,beta=0.5)
       
    # 
    print "{} trained on {} samples.".format(learner.__class__.__name__, sample_size)  
    print "the prediction time is {}".format(results['pred_time'])
    print "the accuracy on train set is {}".format(results['acc_train'])
    print "the accuracy on test set is {}".format(results['acc_test'])
    print "the fscore on train set is {}".format(results['f_train'])
    print "the fscore on test set is {}".format(results['f_test'])
    # 
    return results
